\begin{thm}{214}{\hosi 4}{慶応理工 (2015)}
 $m>0$とし、$f(x)=\dfrac{m^2}{2}\cos 2x-m\cos x$ $(-\pi<x<\pi)$と定める。
 \begin{enumerate}
  \item $f(x)$の最小値を求めよ。
  \item 曲線$y=f(x)$が$x$軸と接するとき、$m$の値を求めよ。
  \item (2)のとき、曲線$y=f(x)$と$x$軸で囲まれた領域を$x$軸のまわりに1回転させて得られる立体の体積を求めよ。
 \end{enumerate}
\end{thm}
