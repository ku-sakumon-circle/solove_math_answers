\begin{thm}{233}{\hosi 2}{阪大 (2020)}
 三角形$\mr{ABC}$において、辺$\mr{AB}$の長さを$c$、辺$\mr{CA}$の長さを$b$で表す。
 \begin{enumerate}
  \item $\angle\mr{ACB}=3\angle\mr{ABC}$であるとき、$c<3b$を示せ。
  \item $n$を2以上の自然数とする。$\angle\mr{ACB}=n\mr{ABC}$であるとき、$c<nb$を示せ。
 \end{enumerate}
 \begin{flushright}
  {\small (1): 文系、(2): 理系}
 \end{flushright}
\end{thm}
