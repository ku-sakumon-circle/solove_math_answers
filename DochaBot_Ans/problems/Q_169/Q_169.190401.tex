\begin{thm}{169}{\hosi 7}{学コン}
 $f(x)=x(2-\log x)$ とする。
 \begin{enumerate}
  \item $1<x<e$ ならば、$2<f(x)<e$ であることを示せ。
  \item $a_1=\alpha$ $(1<\alpha <e)$, $a_{n+1}=f(a_n)$ $(n=1, 2, 3, \cdots)$ により定まる数列$\{a_n\}$について、$\disp \lim_{n\to\infty} a_n = e$ であることを示せ。
 \end{enumerate}
\end{thm}
