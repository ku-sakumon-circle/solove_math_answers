\begin{thm}{206}{\hosi 8}{駿台全国 誘導抜き 改題}
 $n$を正の整数とする。成分がすべて有理数である$2n+1$個のベクトル
 \[ \vvv{v_k}=(x_k,y_k) \quad (k=1, 2, \cdots, 2n+1) \]
 について、以下の2つの関係式
 \begin{align*}
  \vvv{v_1}+\vvv{v_2}+\cdots+\vvv{v_{2n+1}}=\vvv{0} \\
  |\vvv{v_1}|=|\vvv{v_2}|=\cdots=|\vvv{v_{2n+1}}|
 \end{align*}
 を同時に満たすならば、
 \[ \vvv{v_1}=\vvv{v_2}=\cdots=\vvv{v_{2n+1}}=\vvv{0} \]
 であることを証明せよ。
\end{thm}
