\begin{thm}{137}{\hosi 3\maru}{Australia (1999) 改題}
 以下の連立方程式を解け。
 \begin{align*}
  \left\{
  \begin{aligned}
   x+\lfloor y \rfloor + \{z\} &= 33.4 \\
   y+\lfloor z \rfloor + \{x\} &= 34.3 \\
   z+\lfloor x \rfloor + \{y\} &= 43.3 \\
  \end{aligned}
  \right.
 \end{align*}
 ただし、$\lfloor r \rfloor$で$r$以下の最大の整数、$\{r\}$で$r$の小数部分を表す。
\end{thm}
