\begin{thm}{131}{\hosi 6}{東京大 (2007)}
 $x, a$は$0<x<a$を満たす実数であるとする。
 \begin{enumerate}
  \item 以下の不等式を示せ。
	\[ \frac{2x}{a} < \int_{a-x}^{a+x}\! \frac{dt}{t} < x\left(\frac{1}{a-x}+\frac{1}{a+x}\right) \]
  \item $0.68 < \log 2 < 0.71$ であることを証明せよ。ただし、対数は自然対数であるとする。
 \end{enumerate}
\end{thm}
