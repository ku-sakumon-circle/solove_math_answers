\begin{thm}{234}{\hosi 6 \maru}{東京大理系 (2020)}
 $a, b, c, p$を実数とする。不等式、
 \begin{align*}
  ax^2+bx+c&>0 \\
  bx^2+cx+a&>0 \\
  cx^2+ax+b&>0
 \end{align*}
 を全て満たす実数$x$の集合と、$x>p$を満たす実数の集合が一致しているとする。次を示せ。
 \begin{enumerate}
  \item $a, b, c \ge 0$
  \item $a, b, c$ のうち少なくとも1つは0である。
  \item $p=0$
 \end{enumerate}
\end{thm}
