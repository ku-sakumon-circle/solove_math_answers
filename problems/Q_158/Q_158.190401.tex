\begin{thm}{158}{\hosi 2}{河合マーク\Rnum{1}A}
 $a\in\mathbb{N}$, $b\in\mathbb{Z}$, $A=\{3, 4, 7, 9, a^2-a\}$, $B=\{0, 8, a+b+1, 2a+b\}$ とする。
 \begin{enumerate}
  \item $A\cap B = \{3, 4\}$ のとき、$a$と$b$を求めよ。
  \item $A-B = \{2, 3, 7, 9\}$ のとき、$a$と$b$を求めよ。
 \end{enumerate}
\end{thm}
